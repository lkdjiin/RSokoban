def get_action(hash)
  if hash[:type] == :start
    @level_title = hash[:title] 
    @set_title = hash[:set]
    @level_number = hash[:number]
    @set_total = hash[:total]
  end
  display hash
  if [:display, :start].include?(hash[:type])
    ask_player
  else
    # assuming type == 'WIN'
    ask_for_next_level
  end
end
