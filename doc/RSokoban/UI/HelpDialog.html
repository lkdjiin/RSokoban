def initialize(root, title)
  super(root)
  title(title)
  minsize(200, 100)

  text = TkText.new(self) do
    borderwidth 1
    font TkFont.new('times 12 bold')
    grid('row' => 0, 'column' => 0)
  end
  
  help="Welcome to RSokoban !\n\nGoal of Sokoban game is to place each crate on a storage location.\nMove the man using the arrow keys.\nFor a more comprehensive help, please visit the wiki at https://github.com/lkdjiin/RSokoban/wiki.\n"
  
  text.insert 'end', help

  @ok = TkButton.new(self) do
    text 'OK'
    grid('row'=>1, 'column'=>0)
  end
  @ok.command { ok_on_clic }
end
