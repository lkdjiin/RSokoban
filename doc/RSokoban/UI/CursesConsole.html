def get_action(type, level, message)
	if type == 'START' or type == 'END_OF_SET'
		@level_title = message 
		message = 'OK move 0'
		Curses.clear
	end
	if type == 'DISPLAY' or type == 'START'  or type == 'END_OF_SET'
		ask_player level, message
	else
		# assuming type == 'WIN'
		askForNextLevel level, message
	end
end
