def initialize(root, title)
  super(root)
  title(title)
  minsize(200, 100)
  @state = 'CANCEL'
  grab
  
  $spinval = TkVariable.new
  @spin = TkSpinbox.new(self) do
    textvariable($spinval)
    grid('row'=>0, 'column'=>0)
  end
  @spin.to(999)
  @spin.from(1)
  
  @ok = TkButton.new(self) do
    text 'OK'
    grid('row'=>1, 'column'=>0)
  end
  @ok.command { ok_on_clic }
  
  @cancel = TkButton.new(self) do
    text 'Cancel'
    grid('row'=>1, 'column'=>1)
  end
  @cancel.command { cancel_on_clic }
  
  wait_destroy
end
